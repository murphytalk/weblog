应胖子要求草成。




但不能删除目录::


 #!/bin/sh
 #remove files by data - created for YB

 if [ ! -n "$2" ]
 then
     echo "Usage: `basename $0` destination_dir YYYY-MM-DD"
     exit
 fi

 ls $1 --time-style=+%F -l |grep "^-.*$2" |awk -v PATH=$1 -F "$2 " '{print "\\""PATH"/"$2"\\""}'|xargs rm -f
